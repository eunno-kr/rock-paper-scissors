```import random

def 가바보_선택(누구, 리스트_가바보):
    가바보 = None
    if 누구 == '사용자':
        while True:
            가바보 = input("가위, 바위, 보 중에 하나만 입력해 주세요: ")
            if 가바보 in ['가위', '바위', '보']:
                break
            print("사용자가 잘못 선택했어요. 다시 입력해 주세요.")
    else:
        가바보 = 리스트_가바보[random.randint(0,2)]
        print(f"컴퓨터가 선택한 것은 '{가바보}' 입니다.")

    return 가바보

def 승부결정_승횟수누적(사용자_가바보, 컴퓨터_가바보, 승리횟수누적, 사전_가바보_심판):
    승자 = 사전_가바보_심판[사용자_가바보][컴퓨터_가바보]

    if 승자 == '사':
        승리횟수누적['사'] += 1
        print(f"사용자가 이겼어요! (현재 스코어 - 사용자: {승리횟수누적['사']}, 컴퓨터: {승리횟수누적['컴']})")
    elif 승자 == '컴':
        승리횟수누적['컴'] += 1
        print(f"컴퓨터가 이겼어요! (현재 스코어 - 사용자: {승리횟수누적['사']}, 컴퓨터: {승리횟수누적['컴']})")
    else:
        print(f"이번 판은 무승부예요. (현재 스코어 - 사용자: {승리횟수누적['사']}, 컴퓨터: {승리횟수누적['컴']})")

    return 승리횟수누적

print("=== 5전 3선승제 가위바위보 게임을 시작합니다! ===")

승리횟수 = {'사': 0, '컴': 0}

리스트_가바보 = {0:'가위', 1:'바위', 2:'보'}

사전_가바보_심판 = {
    '가위': {'가위': None, '바위': '컴', '보': '사'},
    '바위': {'가위': '사', '바위': None, '보': '컴'},
    '보':   {'가위': '컴', '바위': '사', '보': None}
}

while True:
    print("-" * 40)

    사용자_가바보 = 가바보_선택('사용자', 리스트_가바보)
    컴퓨터_가바보 = 가바보_선택('컴퓨터', 리스트_가바보)

    승리횟수 = 승부결정_승횟수누적(사용자_가바보, 컴퓨터_가바보, 승리횟수, 사전_가바보_심판)

    if 승리횟수['사'] >= 3:
        print("\n 축하합니다! 사용자가 먼저 3승을 달성하여 최종 승리했습니다! ")
        break
    elif 승리횟수['컴'] >= 3:
        print("\n 아쉽네요. 컴퓨터가 먼저 3승을 달성했습니다. 다음 기회에! ")
        break
```
